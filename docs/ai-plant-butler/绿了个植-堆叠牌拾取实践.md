# 绿了个植 - 堆叠牌拾取实践

## 一、核心原理

### 1. 多层堆叠布局
- 每层有行列网格
- 卡牌状态：可拾取（白色）/不可拾取（红色）
- 判断规律：第一层某位置能否拾取，取决于第二层四个对应位置是否有牌

### 2. 数据抽象
- 用数据表格记录每层牌的状态（1=有牌，0=无牌）
- 卡牌被取走后，刷新所有剩余牌的状态

```javascript
const layer1 = [
```

### 2. 判断卡牌是否可拾取

```javascript
function canPick(row, col, layer) {
  if (layer === 1) {
    // 第一层牌：检查第二层对应位置
    const row2 = row * 2
    const col2 = col * 2
    
    // 检查第二层四个对应位置
    if (layer2[row2] && layer2[row2][col2] === 1) return false
    if (layer2[row2] && layer2[row2][col2 + 1] === 1) return false
    if (layer2[row2 + 1] && layer2[row2 + 1][col2] === 1) return false
    if (layer2[row2 + 1] && layer2[row2 + 1][col2 + 1] === 1) return false
    
    return true
  }
  return true // 第二层牌总是可拾取
}
```

### 3. 刷新卡牌状态

```javascript
function refreshAllCards() {
  // 遍历所有卡牌
  for (let row = 0; row < layer1.length; row++) {
    for (let col = 0; col < layer1[row].length; col++) {
      if (layer1[row][col] === 1) {

  for (let row = 0; row < layer2.length; row++) {
    for (let col = 0; col < layer2[row].length; col++) {
      if (layer2[row][col] === 1) {
        updateCardState(row, col, 2, true)
```

## 三、绿了个植适配方案

### 1. 差异分析

| 维度 | 羊了个羊 | 绿了个植 |

| 拾取方式 | 点击选砖放入底部7格槽位 | 滑动相邻消除 |

### 3. 实现方案

```javascript
// 绿了个植的堆叠牌拾取逻辑
class PlantStack {
  constructor() {
    this.layers = [] // 多层堆叠数据
    this.cards = []  // 卡牌对象列表
  }
  
  // 初始化多层布局
  initLayers(config) {
    this.layers = config.layers.map((layerData, layerIndex) => {
      return layerData.map((row, rowIndex) => {
        return row.map((card, colIndex) => {
          return {
            layer: layerIndex,
            row: rowIndex,
            col: colIndex,
            type: card.type,
            state: 'hidden', // hidden, visible, picked
            position: this.calculatePosition(layerIndex, rowIndex, colIndex)
          }
        })
      })
    })
  }
  
  // 检查卡牌是否可滑动
  canSwipe(card) {
    if (card.state !== 'visible') return false
    
    // 检查是否有相邻的相同类型卡牌
    const adjacent = this.getAdjacentCards(card)
    return adjacent.some(adj => adj.type === card.type)
  }
  
  // 获取相邻卡牌
  getAdjacentCards(card) {
    const adjacent = []
    const { layer, row, col } = card
    
    // 检查同一层相邻位置
    const directions = [
      { row: row - 1, col: col },
      { row: row + 1, col: col },
      { row: row, col: col - 1 },
      { row: row, col: col + 1 }
    ]
    
    for (const dir of directions) {
      const adjacentCard = this.getCard(layer, dir.row, dir.col)
      if (adjacentCard && adjacentCard.state === 'visible') {
        adjacent.push(adjacentCard)
      }
    }
    
    return adjacent
  }
  
  // 处理滑动消除
  handleSwipe(card1, card2) {
    if (card1.type !== card2.type) return false
    if (!this.isAdjacent(card1, card2)) return false
    
    // 执行消除
    this.removeCard(card1)
    this.removeCard(card2)
    
    // 刷新状态
    this.refreshAllCards()
    
    return true
  }
  
  // 刷新所有卡牌状态
  refreshAllCards() {
    for (const layer of this.layers) {
      for (const row of layer) {
        for (const card of row) {
          if (card.state === 'hidden') {
            // 检查是否被上层遮挡
            if (!this.isBlocked(card)) {
              card.state = 'visible'
            }
          }
        }
      }
    }
  }
}
```
